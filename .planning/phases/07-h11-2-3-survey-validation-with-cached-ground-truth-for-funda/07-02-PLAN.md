Ground truth generated by tests/generate_ground_truth.py from the original
ExtendedKahlerCone code. See .planning/phases/07-*/07-CONTEXT.md for details.

import json
import pickle
from collections import Counter
from pathlib import Path

import numpy as np
import pytest

from cytools import fetch_polytopes
from cybir.core.ekc import CYBirationalClass
from cybir.core.types import ContractionType

GROUND_TRUTH_DIR = Path(__file__).parent / "ground_truth"
GV_CACHE_DIR = GROUND_TRUTH_DIR / "gv_cache"

# --- Contraction type mapping ---
# cybir has SU2_NONGENERIC_CS and GROSS_FLOP which the original code lacks.
# Map them back for comparison.
_CYBIR_TO_ORIG = {
    "SU2_NONGENERIC_CS": "SU2",
    "GROSS_FLOP": "SYMMETRIC_FLOP",
}

def _map_cybir_type(ct_name):
    """Map cybir contraction type name to ground truth equivalent."""
    return _CYBIR_TO_ORIG.get(ct_name, ct_name)

**Discovery of available ground truth files:**
```python
def _discover_polytopes(h11):
    d = GROUND_TRUTH_DIR / f"h11_{h11}"
    if not d.exists():
        return []
    return sorted(int(p.stem.split("_")[1]) for p in d.glob("poly_*.json"))

def _polytope_params():
    """Generate (h11, poly_idx) pairs for parametrization."""
    params = []
    for h11 in (2, 3):
        for idx in _discover_polytopes(h11):
            params.append(pytest.param((h11, idx), id=f"h11={h11}_poly_{idx}"))
    return params

def _orbit_params():
    """Generate (h11, poly_idx) for polytopes with finite Coxeter (full BFS cached)."""
    params = []
    for h11 in (2, 3):
        for idx in _discover_polytopes(h11):
            gt_path = GROUND_TRUTH_DIR / f"h11_{h11}" / f"poly_{idx}.json"
            with open(gt_path) as f:
                gt = json.load(f)
            if gt.get("full_bfs") is not None:
                params.append(pytest.param((h11, idx), id=f"h11={h11}_poly_{idx}"))
    return params
```

    """Load cached GV Invariants object."""
    path = GV_CACHE_DIR / f"h11_{h11}" / f"poly_{poly_idx}.pkl"

    """Get CYTools CalabiYau for a polytope. Caches polytope lists."""
    if h11 not in _POLYS_CACHE:

**Helper: content-based phase matching (Pitfall 4):**
The original code indexes phases by position. Cybir labels them. Match by content (intersection numbers + c2):
```python
def _match_phases(gt_npz, cybir_phases, prefix=""):
    """Match ground truth phases to cybir phases by content (int_nums + c2).

    Returns dict {gt_phase_idx: cybir_phase_idx} or raises AssertionError.
    """
    n_gt = 0
    while f"{prefix}phase_{n_gt}_int_nums" in gt_npz:
        n_gt += 1

    assert len(cybir_phases) == n_gt, (
        f"Phase count mismatch: ground truth has {n_gt}, cybir has {len(cybir_phases)}"
    )

    matching = {}
    used = set()
    for gt_idx in range(n_gt):
        gt_int = gt_npz[f"{prefix}phase_{gt_idx}_int_nums"]
        gt_c2 = gt_npz[f"{prefix}phase_{gt_idx}_c2"]
        found = False
        for ci, phase in enumerate(cybir_phases):
            if ci in used:
                continue
            if np.array_equal(phase.int_nums, gt_int) and np.array_equal(phase.c2, gt_c2):
                matching[gt_idx] = ci
                used.add(ci)
                found = True
                break
        assert found, f"Ground truth phase {gt_idx} not found in cybir phases"
    return matching
```

**Helper: compare generator sets:**
```python
def _compare_gen_sets(expected, actual, name):
    """Compare two sets of integer tuples."""
    exp_set = {tuple(int(x) for x in g) for g in expected}
    act_set = {tuple(int(x) for x in g) for g in actual}
    assert exp_set == act_set, (
        f"{name} mismatch:\n"
        f"  Missing from cybir: {sorted(exp_set - act_set)}\n"
        f"  Extra in cybir: {sorted(act_set - exp_set)}"
    )
```

**Test function 1: `test_fundamental_domain`**
```python
@pytest.mark.slow
@pytest.mark.parametrize("h11_idx", _polytope_params())
def test_fundamental_domain(h11_idx):
    h11, poly_idx = h11_idx
    gt_dir = GROUND_TRUTH_DIR / f"h11_{h11}"
    with open(gt_dir / f"poly_{poly_idx}.json") as f:
        gt_json = json.load(f)
    gt_npz = np.load(gt_dir / f"poly_{poly_idx}.npz")

    # Run cybir
    cy = _get_cy(h11, poly_idx)
    gvs = _load_gvs(h11, poly_idx)
    ekc = CYBirationalClass.from_gv(cy, max_deg=10, verbose=False, gvs=gvs)

    # 1. Phase count
    assert len(ekc.phases) == gt_json["n_phases"], (
        f"Phase count: expected {gt_json['n_phases']}, got {len(ekc.phases)}"
    )

    # 2. Content-based phase matching + per-phase data
    matching = _match_phases(gt_npz, ekc.phases)
    for gt_idx, cybir_idx in matching.items():
        phase = ekc.phases[cybir_idx]
        # Intersection numbers (exact integer)
        gt_int = gt_npz[f"phase_{gt_idx}_int_nums"]
        assert np.array_equal(phase.int_nums, gt_int), (
            f"Phase {gt_idx} int_nums mismatch"
        )
        # c2 (exact integer)
        gt_c2 = gt_npz[f"phase_{gt_idx}_c2"]
        assert np.array_equal(phase.c2, gt_c2), (
            f"Phase {gt_idx} c2 mismatch"
        )
        # Kahler cone rays (set comparison, sorted)
        gt_rays_key = f"phase_{gt_idx}_kahler_rays"
        if gt_rays_key in gt_npz:
            gt_rays = {tuple(int(x) for x in r) for r in gt_npz[gt_rays_key]}
            cybir_rays = {tuple(int(x) for x in r) for r in phase.kahler_cone.rays()}
            assert gt_rays == cybir_rays, (
                f"Phase {gt_idx} Kahler rays mismatch"
            )

    # 3. Contraction type counts (mapped)
    cybir_counts = Counter()
    for c in ekc.contractions:
        if c.contraction_type is not None:
            cybir_counts[_map_cybir_type(c.contraction_type.name)] += 1
    gt_counts = Counter(gt_json["contraction_type_counts"])
    assert cybir_counts == gt_counts, (
        f"Contraction type counts mismatch:\n"
        f"  Expected: {dict(gt_counts)}\n"
        f"  Got: {dict(cybir_counts)}"
    )

    # 4. Infinity cone generators
    _compare_gen_sets(
        gt_json["infinity_cone_gens"],
        ekc.infinity_cone_gens,
        "Infinity cone gens"
    )

    # 5. Effective cone generators
    _compare_gen_sets(
        gt_json["eff_cone_gens"],
        ekc.eff_cone_gens,
        "Effective cone gens"
    )

    gt_npz.close()
```

@pytest.mark.slow

def test_orbit_expansion(h11_idx):
    h11, poly_idx = h11_idx
    gt_dir = GROUND_TRUTH_DIR / f"h11_{h11}"
    with open(gt_dir / f"poly_{poly_idx}.json") as f:
        gt_json = json.load(f)
    gt_npz = np.load(gt_dir / f"poly_{poly_idx}.npz")
    full_bfs = gt_json["full_bfs"]

    # Run cybir with orbit expansion
    cy = _get_cy(h11, poly_idx)
    gvs = _load_gvs(h11, poly_idx)
    ekc = CYBirationalClass.from_gv(cy, max_deg=10, verbose=False, gvs=gvs)

    # 2. Content-based phase matching for full orbit
    matching = _match_phases(gt_npz, ekc.phases, prefix="full_")
