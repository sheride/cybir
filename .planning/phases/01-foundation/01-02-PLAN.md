```python
import numpy as np
import hsnf

def charge_matrix_hsnf(vectors):
    """Compute integer relations among vectors via Smith Normal Form.

    Args:
        vectors: List of integer vectors, shape (n_vectors, dim).

    Returns:
        Integer relation matrix, shape (n_relations, n_vectors).
        Rows are integer linear relations among the input vectors.
    """
    D, U, W = hsnf.smith_normal_form(np.array(vectors).T)
    rank = sum(1 for i in range(min(len(D), len(D[0]))) if D[i, i] != 0)
    relations = W[:, rank:]
    return relations.T
```

**moving_cone** (port from misc.py per D-10):
```python
import cytools

def moving_cone(Q, verbose=False):
    """Compute the moving cone from the charge matrix Q.

    Iterates over columns of Q, computes the cone of remaining columns'
    hyperplanes, and forms their intersection.

    Args:
        Q: Charge matrix, shape (n_relations, n_divisors).
        verbose: If True, print progress.

    Returns:
        cytools.Cone: The moving cone.
    """
    hyps = np.vstack([
        cytools.Cone(rays=np.delete(Q, i, axis=1).T).hyperplanes()
        for i in range(Q.shape[1])
    ])
    return cytools.Cone(hyperplanes=hyps)
```

def tuplify(arr):
    """Convert a numpy ndarray into a nested tuple structure."""
    if isinstance(arr, np.ndarray):
        return tuple(tuplify(x) for x in arr.tolist())
    if isinstance(arr, list):
        return tuple(tuplify(x) for x in arr)
    return arr
```

def normalize_curve(curve, return_sign=False):
    """Normalize a curve class so first nonzero element is positive.

    Args:
        curve: numpy array representing a curve class.
        return_sign: If True, also return the sign factor.

    Returns:
        Normalized curve as tuple, or (tuple, sign) if return_sign=True.
    """
    if next(c for c in curve if c != 0) > 0:
        to_return = tuple(curve.tolist())
        sign = 1
    else:
        to_return = tuple((-curve).tolist())
        sign = -1
    return (to_return, sign) if return_sign else to_return
```

def projection_matrix(curve):
    """Return (N-1) x N matrix projecting onto complement of curve.

    Uses Smith Normal Form to find a basis for the sublattice
    orthogonal to the given curve class.

    Args:
        curve: 1D numpy array representing a curve class.

    Returns:
        numpy array of shape (N-1, N).
    """
    return hsnf.smith_normal_form(np.array([curve]).T)[1][1:]

```python
"""Phase adjacency graph for extended Kahler cone construction."""

import networkx as nx
from .types import CalabiYauLite, ExtremalContraction


class PhaseGraph:
    """Adjacency graph with CalabiYauLite phases as nodes and
    ExtremalContraction objects as edges.

    Uses an undirected networkx.Graph internally. Contractions connect
    two phases symmetrically (you can flop in either direction). The
    start/end distinction is stored on the ExtremalContraction object,
    not the graph edge.
    """

    def __init__(self):
        self._graph = nx.Graph()

    def add_phase(self, phase):
        """Add a phase node.

        Args:
            phase: CalabiYauLite with a label set.
        """
        self._graph.add_node(phase.label, phase=phase)

    def add_contraction(self, contraction):
        """Add a contraction edge between two phases.

        Args:
            contraction: ExtremalContraction with start_phase and
                end_phase labels set.
        """
        self._graph.add_edge(
            contraction.start_phase,
            contraction.end_phase,
            contraction=contraction)

    @property
    def phases(self):
        """All phase objects."""
        return [d['phase'] for _, d in self._graph.nodes(data=True)]

    @property
    def contractions(self):
        """All contraction objects."""
        return [d['contraction'] for _, _, d in self._graph.edges(data=True)]

    @property
    def num_phases(self):
        """Number of phases in the graph."""
        return self._graph.number_of_nodes()

    @property
    def num_contractions(self):
        """Number of contractions in the graph."""
        return self._graph.number_of_edges()

    def neighbors(self, label):
        """Phases adjacent to the given phase.

        Args:
            label: Phase label string.

        Returns:
            List of CalabiYauLite objects adjacent to the given phase.
        """
        return [self._graph.nodes[n]['phase']
                for n in self._graph.neighbors(label)]

    def get_phase(self, label):
        """Get a phase by label.

        Args:
            label: Phase label string.

        Returns:
            CalabiYauLite object.
        """
        return self._graph.nodes[label]['phase']

    def __repr__(self):
        return (f"PhaseGraph(phases={self.num_phases}, "
                f"contractions={self.num_contractions})")
```
