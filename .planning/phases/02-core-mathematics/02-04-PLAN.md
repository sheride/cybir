```python
def flop(self, curve, gv_series, label=None):
    """Create a new phase by flopping across a wall.

    Thin wrapper around :func:`cybir.core.flop.flop_phase`.

    Parameters
    ----------
    curve : numpy.ndarray
        Flopping curve class.
    gv_series : list[int]
        GV series for the flopping curve.
    label : str, optional
        Label for the new phase.

    Returns
    -------
    CalabiYauLite
        New phase with transformed intersection numbers and c2.
    """
    from .flop import flop_phase
    return flop_phase(self, curve, gv_series, label=label)
```

Use lazy imports (`from .flop import ...` inside the method body) to avoid circular import issues. The methods are:

- `CalabiYauLite.flop(self, curve, gv_series, label=None)` -> delegates to `flop.flop_phase`

The standalone `classify_contraction` function is the correct API. Convenience wrappers would just obscure the required inputs.

```python
"""Integration tests: verify cybir math against original code snapshots."""

import json
import pathlib
import numpy as np
import pytest

from cybir.core.flop import wall_cross_intnums, wall_cross_c2
from cybir.core.gv import compute_gv_eff
from cybir.core.classify import classify_contraction
from cybir.core.types import ContractionType

FIXTURES_DIR = pathlib.Path(__file__).parent / "fixtures" / "h11_2"

# Map original category strings to ContractionType
CATEGORY_MAP = {
    "asymptotic": ContractionType.ASYMPTOTIC,
    "CFT": ContractionType.CFT,
    "su(2) enhancement": ContractionType.SU2,
    "symmetric flop": ContractionType.SYMMETRIC_FLOP,
    "generic flop (I)": ContractionType.FLOP,
    "generic flop (II)": ContractionType.FLOP,
}


def load_fixtures():
    """Load all polytope fixture files."""
    fixtures = []
    if not FIXTURES_DIR.exists():
        return fixtures
    for f in sorted(FIXTURES_DIR.glob("polytope_*.json")):
        with open(f) as fp:
            data = json.load(fp)
        fixtures.append(data)
    return fixtures


@pytest.fixture(params=load_fixtures(), ids=lambda d: f"polytope_{d['polytope_id']}")
def polytope_data(request):
    return request.param
```

Tests:

- `test_gv_eff_matches_snapshot(polytope_data)`: Verify `compute_gv_eff` gives matching `gv_eff_1` and `gv_eff_3`.
